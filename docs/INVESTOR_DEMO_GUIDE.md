- Check that beta gate is enabled in app

### If campaigns don't appear:
- Verify test data is seeded (run setup script if needed)
- Check that campaigns are in "active" status
- Ensure restaurant is claimed and verified

### If creator profile doesn't load:
- Verify creator onboarding was completed
- Check that portfolio images uploaded successfully
- Ensure creator_profiles table has data

### If restaurant claiming doesn't work:
- Make sure you're claiming a test restaurant (check for `.test` domain or test restaurant names)
- Verify the restaurant is unclaimed (see debugging queries below)

## üîß Debugging & Reset Queries (For Developer Use)

### Debugging: Check Test Data State

```sql
-- 1. Check all prod-* test accounts and their account types
SELECT 
  u.id,
  u.email,
  u.account_type,
  u.is_creator,
  u.is_restaurant,
  CASE WHEN cp.id IS NOT NULL THEN 'Yes' ELSE 'No' END as has_creator_profile,
  CASE WHEN bp.id IS NOT NULL THEN 'Yes' ELSE 'No' END as has_business_profile
FROM users u
LEFT JOIN creator_profiles cp ON cp.user_id = u.id
LEFT JOIN business_profiles bp ON bp.user_id = u.id
WHERE u.email LIKE 'prod-%@bypass.com'
ORDER BY u.email;

-- 2. Check creator profiles status
SELECT 
  u.email,
  cp.display_name,
  cp.bio,
  cp.location,
  cp.specialties,
  cp.open_to_collabs,
  cp.availability_status,
  COUNT(cpi.id) as portfolio_items_count
FROM creator_profiles cp
JOIN users u ON cp.user_id = u.id
LEFT JOIN creator_portfolio_items cpi ON cpi.creator_profile_id = cp.id
WHERE u.email LIKE 'prod-creator%@bypass.com'
GROUP BY u.email, cp.id, cp.display_name, cp.bio, cp.location, cp.specialties, cp.open_to_collabs, cp.availability_status
ORDER BY u.email;

-- 3. Check business profiles and claimed restaurants
SELECT 
  u.email,
  bp.restaurant_id,
  r.name as restaurant_name,
  r.is_claimed,
  r.claimed_by,
  r.website
FROM business_profiles bp
JOIN users u ON bp.user_id = u.id
LEFT JOIN restaurants r ON bp.restaurant_id = r.id
WHERE u.email LIKE 'prod-business%@bypass.com'
ORDER BY u.email;

-- 4. Check unclaimed test restaurants (for claiming test)
SELECT 
  r.id,
  r.name,
  r.city,
  r.state,
  r.website,
  r.is_claimed,
  CASE WHEN r.website LIKE '%.test' THEN 'Test Restaurant' ELSE 'Regular' END as restaurant_type
FROM restaurants r
WHERE r.is_claimed = false
  AND (r.website LIKE '%.test' OR r.name LIKE '%Test%' OR r.name LIKE '%Prod%')
ORDER BY r.name
LIMIT 10;

-- 5. Check active campaigns
SELECT 
  c.id,
  c.title,
  c.status,
  c.budget_cents,
  c.end_date,
  r.name as restaurant_name,
  u.email as business_email,
  COUNT(ca.id) as application_count
FROM campaigns c
JOIN restaurants r ON c.restaurant_id = r.id
JOIN business_profiles bp ON bp.restaurant_id = r.id
JOIN users u ON bp.user_id = u.id
LEFT JOIN campaign_applications ca ON ca.campaign_id = c.id
WHERE c.status = 'active'
  AND u.email LIKE 'prod-business%@bypass.com'
GROUP BY c.id, c.title, c.status, c.budget_cents, c.end_date, r.name, u.email
ORDER BY c.created_at DESC;

-- 6. Check campaign applications
SELECT 
  ca.id,
  ca.status,
  ca.proposed_rate_cents,
  c.title as campaign_title,
  u_creator.email as creator_email,
  u_business.email as business_email
FROM campaign_applications ca
JOIN campaigns c ON ca.campaign_id = c.id
JOIN creator_profiles cp ON ca.creator_id = cp.id
JOIN users u_creator ON cp.user_id = u_creator.id
JOIN restaurants r ON c.restaurant_id = r.id
JOIN business_profiles bp ON bp.restaurant_id = r.id
JOIN users u_business ON bp.user_id = u_business.id
WHERE u_creator.email LIKE 'prod-creator%@bypass.com'
   OR u_business.email LIKE 'prod-business%@bypass.com'
ORDER BY ca.applied_at DESC;

-- 7. Check campaign invitations
SELECT 
  ci.id,
  ci.status,
  ci.message,
  ci.invited_at,
  ci.expires_at,
  c.title as campaign_title,
  u_creator.email as creator_email,
  u_business.email as business_email
FROM campaign_invitations ci
JOIN campaigns c ON ci.campaign_id = c.id
JOIN creator_profiles cp ON ci.creator_id = cp.id
JOIN users u_creator ON cp.user_id = u_creator.id
JOIN users u_business ON ci.invited_by = u_business.id
WHERE u_creator.email LIKE 'prod-creator%@bypass.com'
   OR u_business.email LIKE 'prod-business%@bypass.com'
ORDER BY ci.invited_at DESC;
```

### Reset: Clean Test Data for Fresh Testing

```sql
-- ‚ö†Ô∏è WARNING: This will delete all test data. Use with caution!
-- Run this to reset test state between demo runs

BEGIN;

-- 1. Delete campaign invitations
DELETE FROM campaign_invitations ci
WHERE EXISTS (
  SELECT 1 FROM campaigns c
  JOIN restaurants r ON c.restaurant_id = r.id
  JOIN business_profiles bp ON bp.restaurant_id = r.id
  JOIN users u ON bp.user_id = u.id
  WHERE c.id = ci.campaign_id
    AND u.email LIKE 'prod-%@bypass.com'
)
OR EXISTS (
  SELECT 1 FROM creator_profiles cp
  JOIN users u ON cp.user_id = u.id
  WHERE cp.id = ci.creator_id
    AND u.email LIKE 'prod-%@bypass.com'
);

-- 2. Delete deliverables
DELETE FROM campaign_deliverables cd
WHERE EXISTS (
  SELECT 1 FROM campaign_applications ca
  JOIN campaigns c ON ca.campaign_id = c.id
  JOIN restaurants r ON c.restaurant_id = r.id
  JOIN business_profiles bp ON bp.restaurant_id = r.id
  JOIN users u ON bp.user_id = u.id
  WHERE ca.id = cd.campaign_application_id
    AND u.email LIKE 'prod-%@bypass.com'
)
OR EXISTS (
  SELECT 1 FROM campaign_applications ca
  JOIN creator_profiles cp ON ca.creator_id = cp.id
  JOIN users u ON cp.user_id = u.id
  WHERE ca.id = cd.campaign_application_id
    AND u.email LIKE 'prod-%@bypass.com'
);

-- 3. Delete campaign applications
DELETE FROM campaign_applications ca
WHERE EXISTS (
  SELECT 1 FROM campaigns c
  JOIN restaurants r ON c.restaurant_id = r.id
  JOIN business_profiles bp ON bp.restaurant_id = r.id
  JOIN users u ON bp.user_id = u.id
  WHERE c.id = ca.campaign_id
    AND u.email LIKE 'prod-%@bypass.com'
)
OR EXISTS (
  SELECT 1 FROM creator_profiles cp
  JOIN users u ON cp.user_id = u.id
  WHERE cp.id = ca.creator_id
    AND u.email LIKE 'prod-%@bypass.com'
);

-- 4. Delete campaigns
DELETE FROM campaigns c
WHERE EXISTS (
  SELECT 1 FROM restaurants r
  JOIN business_profiles bp ON bp.restaurant_id = r.id
  JOIN users u ON bp.user_id = u.id
  WHERE r.id = c.restaurant_id
    AND u.email LIKE 'prod-%@bypass.com'
);

-- 5. Unclaim restaurants (but keep restaurant data)
UPDATE restaurants r
SET is_claimed = false,
    claimed_by = NULL
WHERE EXISTS (
  SELECT 1 FROM business_profiles bp
  JOIN users u ON bp.user_id = u.id
  WHERE bp.restaurant_id = r.id
    AND u.email LIKE 'prod-%@bypass.com'
);

-- 6. Delete business profiles
DELETE FROM business_profiles bp
WHERE EXISTS (
  SELECT 1 FROM users u
  WHERE u.id = bp.user_id
    AND u.email LIKE 'prod-%@bypass.com'
);

-- 7. Delete creator portfolio items
DELETE FROM creator_portfolio_items cpi
WHERE EXISTS (
  SELECT 1 FROM creator_profiles cp
  JOIN users u ON cp.user_id = u.id
  WHERE cp.id = cpi.creator_profile_id
    AND u.email LIKE 'prod-%@bypass.com'
);

-- 8. Delete creator profiles
DELETE FROM creator_profiles cp
WHERE EXISTS (
  SELECT 1 FROM users u
  WHERE u.id = cp.user_id
    AND u.email LIKE 'prod-%@bypass.com'
);

-- 9. Reset user account types (but keep users)
UPDATE users
SET account_type = 'consumer',
    is_creator = false,
    is_restaurant = false
WHERE email LIKE 'prod-%@bypass.com';

COMMIT;

-- Verify reset
```
